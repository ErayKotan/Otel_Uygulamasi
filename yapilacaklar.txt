create proc sp_indirimsorgu
(
@gelentarih datetime
)
as
begin
if exists (select KampanyaIndirimOran from tbl_Kampanyalar where KampanyaBaslangicZaman<@gelentarih and KampanyaBitisTarihi>@gelentarih)
begin
select KampanyaIndirimOran from tbl_Kampanyalar where KampanyaBaslangicZaman<@gelentarih and KampanyaBitisTarihi>@gelentarih order by KampanyaIndirimOran desc
end
else
begin
return '0'
end
end
